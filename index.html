// ADMIN CONTROL const isAdmin = true; // DATA let sections = ["Haldi", "Mehendi", "Muhurtham"]; let posts = []; const sectionsContainer = document.getElementById("sectionsContainer"); const postsContainer = document.getElementById("postsContainer"); const cameraBtn = document.getElementById("cameraBtn"); const cameraInput = document.getElementById("cameraInput"); // LOAD SECTIONS function loadSections() { sectionsContainer.innerHTML = ""; sections.forEach(section => { const div = document.createElement("div"); div.className = "story"; div.innerHTML = `
${section}
`; div.onclick = () => filterPosts(section); sectionsContainer.appendChild(div); }); } // LOAD POSTS function loadPosts(data = posts) { postsContainer.innerHTML = ""; data.forEach(post => { const div = document.createElement("div"); div.className = "post"; div.innerHTML = `
${post.section}
`; postsContainer.appendChild(div); }); } // FILTER function filterPosts(section) { const filtered = posts.filter(p => p.section === section); loadPosts(filtered); } // CAMERA if (!isAdmin) { cameraBtn.style.display = "none"; } cameraBtn.onclick = () => { cameraInput.click(); }; cameraInput.onchange = () => { const file = cameraInput.files[0]; if (!file) return; let section = prompt("Enter section name"); if (!section) return; if (!sections.includes(section)) { sections.push(section); loadSections(); } const reader = new FileReader(); reader.onload = function(e) { posts.unshift({ section: section, image: e.target.result }); loadPosts(); }; reader.readAsDataURL(file); }; // INIT loadSections(); loadPosts();
